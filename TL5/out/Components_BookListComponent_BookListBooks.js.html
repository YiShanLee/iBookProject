import React, { Component } from "react";

import Book from "./Book";

/**
 *  @class BookListBooks - Represents the part of the table in which the books are listed.
 *  @extends Component
*/
class BookListBooks extends Component {

    /** Renders the books of the bookslist in table rows.
     * 
     * @returns {Component} - The BookListBooks component.
     */
    render(){
        
        /**
        * @constant  
        * @type {Object}
        */  
        const books = this.props.books;

        return (
            <tbody id="book-list">
                {books.map((book) => (
                    <Book key={book.isbn} book={book}
                    updateBook={(event) => {
                    this.props.handleUpdate(event);
                    }}
                    deleteBook={(isbn) => {
                    this.props.handleDelete(isbn);
                    }}/>
                ))}
            </tbody>
        );
    }
}

/**
 * @exports BookListBooks
 */
export default BookListBooks;
